#include "stdafx.h"
#include<iostream>
using namespace std;
#include <string>

string M = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";

int main() {

	string s;
	cout << "Nhap chuoi : ";
	getline(cin, s);
	int size = s.size();
//Ma
	for (int i = 0; i < 25; i++) {

		s.push_back(M[i]);
	}

	int dem = 0;
	char matrix[5][5];
	for (int i = 0; i < 5; i++) {

		for (int j = 0; j < 5; j++) {

			matrix[i][j] = s[dem];
			dem++;
		}
	}

	for (int i = 0; i < 5; i++) {

		for (int j = 0; j < 5; j++) {

			cout << matrix[i][j] << " ";
		}
		cout << endl;
	}

	string mahoa;
	for (int j = 0; j < 5; j++) {

		for (int i = 0; i < 5; i++) {

			mahoa.push_back(matrix[i][j]);
		}
	}
	cout << "Chuoi ma hoa: " << mahoa << endl;
//Giai.
	dem = 0;
	char matrixG[5][5];
	for (int i = 0; i < 5; i++) {

		for (int j = 0; j < 5; j++) {

			matrixG[i][j] = mahoa[dem];
			dem++;
		}
	}
	string giaima;

	for (int j = 0; j < 5; j++) {

		for (int i = 0; i < 5; i++) {

			giaima.push_back(matrixG[i][j]);
		}
	}
	cout << "Chuoi giai ma: " << giaima << endl;
		system("pause");

}
