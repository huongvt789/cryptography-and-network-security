#include "stdafx.h"
#include<iostream>
using namespace std;
#include <string>
string M = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
string loaiBoKyTuTrungLap(string &s) {
	for (int i = 0; i < s.size(); i++)
		for (int j = i + 1; j < s.size(); j++)
			if (s[j] == s[i]) {
				s.erase(j, 1);
				j--;
			}
	return s;
}
string themChuoiConLaiVao(string &s) {
	for (int i = 0; i < s.size(); i++)
		for (int j = 0; j < M.size(); j++)
			if (M[j] == s[i])
			{
				M.erase(j, 1); j--;
			}
	s = s + M;
	return s;
}
