- Lập trình khôi phục lại chuỗi ban đầu.
*/
#include "stdafx.h"
#include<iostream>
using namespace std;
#include <string>


int main()
{
	string s;
	cout << "Moi nhap chuoi:";
	cin >> s;
	for (int i = 0; i < s.size(); i++) {
		s[i] = s[i] + 3;
	}
	cout << "Chuoi vua nhap la:" << s <<'\n';
	for (int i = 0; i < s.size(); i++)
	{
		s[i] = s[i] - 3;
	}
	cout << "Chuoi sau khi ma hoa la:" << s;
	system("pause");

    return 0;
}
